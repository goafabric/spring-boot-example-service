#deployment
docker tag goafabric/spring-boot-exampleservice goafabric/spring-boot-exampleservice:stable

docker push goafabric/spring-boot-exampleservice:stable
|
docker pull goafabric/spring-boot-exampleservice:stable

#run
docker run -p 8080:8080 --name exampleservice -t goafabric/spring-boot-exampleservice:stable

docker logs -f c78d3bd105c2

#create
docker container create -p 8080:8080 --name exampleservice -t goafabric/spring-boot-exampleservice:stable
docker container run exampleservice

#swarm
docker swarm init

docker stack deploy -c docker-compose.yml exampleservice

docker service logs -f exampleservice_web


#portainer
docker volume create portainer_data
docker run -d -p 9000:9000 --restart always -v /var/run/docker.sock:/var/run/docker.sock -v portainer_data:/data portainer/portainer

#postgres
docker volume create postgres_data

docker run -e POSTGRES_PASSWORD=postgres -d -p 5432:5432 -v postgres_data:/var/lib/postgresql/data postgres:9.6
